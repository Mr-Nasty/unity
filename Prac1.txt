using System;
class Program
{
    static void Main()
    {
        Console.Write("Enter the value of n: ");
        int n = int.Parse(Console.ReadLine());
        int sum = 0;
        for (int i = 1; i <= n; i++)
        {
            sum += i;
        }
        Console.WriteLine("The sum of first " + n + " numbers is: " + sum);
    }
}

using System;
public class HelloWorld
{
    public static void Main(string[] args)
    {
        char grade = 'A';
        switch (grade)
        {
            case 'A':
                Console.WriteLine("Excellent!");
                break;
            case 'B':
                Console.WriteLine("Good job!");
                break;
            case 'C':
                Console.WriteLine("Fair.");
                break;
            case 'D':
                Console.WriteLine("Needs improvement.");
                break;
            case 'F':
                Console.WriteLine("Failed.");
                break;
            default:
                Console.WriteLine("Invalid grade.");
                break;
        }
    }
}



using System;
class Program
{
    static void Main()
    {
        int start = 1;
        int end = 10;
        int sum = 0;
        int i = start;
        while (i <= end)
        {
            if (i % 2 == 0)
            {
                sum += i;
            }
            i++;
        }
        Console.WriteLine("Sum of even numbers = " + sum);
    }
}
