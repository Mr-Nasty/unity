using System;
class Program
{
    static void Main()
    {
        int start = 1;
        int end = 10;
        int sum = 0;
        int i = start;
        while (i <= end)
        {
            if (i % 2 == 0)
            {
                sum += i;
            }
            i++;
        }
        Console.WriteLine("Sum of even numbers = " + sum);
    }
}
