- Player Movement Script (Jump & Move)
- Attach this script to the Circle object.
- Create script: JumpnMove

using UnityEngine;

public class JumpnMove : MonoBehaviour
{
    public float jumpforce = 2f;
    public float speed = 2f;

    public Rigidbody2D rb;

    void Start()
    {
        rb = GetComponent<Rigidbody2D>();
    }

    void Update()
    {
        float moveX = Input.GetAxis("Horizontal");

        rb.velocity = new Vector2(moveX * speed, rb.velocity.y);

        if (Input.GetKeyDown(KeyCode.Space))
        {
            rb.velocity = new Vector2(rb.velocity.x, jumpforce);
        }
    }
}

- Select Circle
- Add Component → JumpnMove
Make sure it also has:
- Rigidbody2D
- CircleCollider2D

Sound Trigger Effector


- Create Square
- Rename: SoundZone
- Add:
Box Collider 2D
Audio Source

- Enable:
Is Trigger
   
- Create script:
AudioEffector2D

using UnityEngine;

public class AudioEffector2D : MonoBehaviour
{
    public AudioSource audioSource;
    private int playerCount = 0;

    void Awake()
    {
        if (audioSource == null)
            audioSource = GetComponent<AudioSource>();
    }

    void OnTriggerEnter2D(Collider2D other)
    {
        if (!other.CompareTag("Player")) return;

        playerCount++;

        if (playerCount == 1 && !audioSource.isPlaying)
            audioSource.Play();
    }

    void OnTriggerExit2D(Collider2D other)
    {
        if (!other.CompareTag("Player")) return;

        playerCount--;

        if (playerCount <= 0)
        {
            playerCount = 0;
            audioSource.Stop();
        }
    }
}

- Select SoundZone
- Add Component → AudioEffector2D, Box Collider 2D, Audio Source
- Then drag audio clip into: Audio Source → AudioClip
AudioEffector2D script
- Select Circle (player)
- In Inspector:
Tag → Player
- Drag an audio clip into the AudioSource.

When player enters the zone → sound plays.
When player leaves → sound stops.
