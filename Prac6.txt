SETUP
- Open Unity project
- Create two scenes and save them as:
    MainScene
    Prac5
- Add both scenes to Build Settings:
    File → Build Profiles → Scene List
    → Click: Add Open Scenes
    → Make sure both scenes are checked
 
 
1. LOADING SCENE
Loads a new scene and replaces the current one.
 
- GameObject → UI → Button
- Create script: load
- Attach to Button
 
using UnityEngine;
using UnityEngine.SceneManagement;
public class load : MonoBehaviour
{
    public void LoadScene()
    {
        SceneManager.LoadScene("Prac5");
    }
}
 
- Select Button
- In Button Inspector → OnClick() → Click +
- Drag Button into the field
- Select: load → LoadScene()
 
 
2. ASYNCHRONOUS LOADING
Loads a scene in the background without freezing the game.
 
- Keep the same Button
- Create script: asyncload
- Attach to Button
 
using UnityEngine;
using UnityEngine.SceneManagement;
using System.Collections;
public class asyncload : MonoBehaviour
{
    public void StartLoading()
    {
        StartCoroutine(LoadAsync());
    }
    IEnumerator LoadAsync()
    {
        AsyncOperation operation =
            SceneManager.LoadSceneAsync("Prac5");
        while (!operation.isDone)
        {
            Debug.Log("Loading: " + (operation.progress * 100) + "%");
            yield return null;
        }
    }
}
 
- Select Button → OnClick() → Click +
- Drag Button into the field
- Select: asyncload → StartLoading()
 
 
3. ADDITIVE LOADING
Loads a new scene on top of the current one (both run at the same time).
 
- Create script: addload
- Attach to Button
 
using UnityEngine;
using UnityEngine.SceneManagement;
public class addload : MonoBehaviour
{
    public void LoadSceneAdd()
    {
        SceneManager.LoadScene("Prac5", LoadSceneMode.Additive);
    }
}
 
- Select Button → OnClick() → Click +
- Drag Button into the field
- Select: addload → LoadSceneAdd()
 
 
4. UNLOADING SCENE
Removes a scene that was loaded additively.
NOTE: This only works on scenes loaded with Additive mode.
      Run Additive Loading first, then Unload.
 
- Create script: unload
- Attach to Button
 
using UnityEngine;
using UnityEngine.SceneManagement;
public class unload : MonoBehaviour
{
    public void UnloadScene()
    {
        SceneManager.UnloadSceneAsync("Prac5");
    }
}
 
- Select Button → OnClick() → Click +
- Drag Button into the field
- Select: unload → UnloadScene()
