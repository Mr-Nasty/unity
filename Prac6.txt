2. Asynchronous Loading

Asynchronous loading loads a scene without freezing the game.

- Keep the same button.
- Create script: asyncload

using UnityEngine;
using UnityEngine.SceneManagement;
using System.Collections;

public class asyncload : MonoBehaviour
{
    public void StartLoading()
    {
        StartCoroutine(LoadAsync());
    }

    IEnumerator LoadAsync()
    {
        AsyncOperation operation =
            SceneManager.LoadSceneAsync("Prac5");

        while (!operation.isDone)
        {
            Debug.Log("Loading : " + (operation.progress * 100) + "%");
            yield return null;
        }
    }
}

- Select Button < Add Component → asyncload

- OnClick() -> asyncload → StartLoading()

4. Unloading Scene

This removes a loaded scene.

- Create script: unload

using UnityEngine;
using UnityEngine.SceneManagement;

public class unload : MonoBehaviour
{
    public void UnloadScene()
    {
        SceneManager.UnloadSceneAsync("Prac5");
    }
}

- Select Button
- Add Component → unload script
- OnClick()
- Select: unload → UnloadScene()
